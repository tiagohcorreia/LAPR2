 public class NetworkManagerUI implements Runnable {
 
 
     @Override

         //GuiHandler.myLaunch(ListDealsGUI.class);

         options.add(new MenuItem("List All Employees", new ListAllEmplyeesUI()));

             }
         } while (option != -1);
     }
